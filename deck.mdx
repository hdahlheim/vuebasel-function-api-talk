```js 1:7 subtitle="everything returned from 'setup' is available in render"
export default {
  setup () {
    return {
      // ...
    }
  }
}
```

```js 1:7 subtitle="'this' is not availably in setup"
export default {
  setup () {
    return {
      // ...
    }
  }
}
```


```js 1:7 subtitle="'setup' takes two arguments"
export default {
  setup (props, context) {
    return {
      // ...
    }
  }
}
```

```js 1:7 subtitle="'props' gives access to all props defined on the component"
export default {
  props:{
    msg: String
  },
  setup (props, context) {
    console.log(props.msg)
  }
}
```

```js 3:8 subtitle="'context' exposes everything else"
export default {
  setup (props, context) {
    // context.attrs
    // context.slots
    // context.refs
    // context.emit
    // context.parent
    // context.root
  }
}
```

```jsx title="Vue 2 component"
<script>
```
```jsx 1:32
import { value, onMounted, onUnmounted } from 'vue'
```

```jsx
function useMousePosition () {
  const x = value(0)
  const y = value(0)

  const updateMouse = (e) => {
    x.value = e.pageX
    y.value = e.pageY
  }

  onMounted(() => {
    window.addEventListener('mousemove', updateMouse)
  })

  onUnmounted(() => {
    window.removeEventListener('mousemove', updateMouse)
  })

  return { x, y }
}

export default {
  setup () {
    return {
      ...useMousePosition()
    }
  }
}
</script>
```
